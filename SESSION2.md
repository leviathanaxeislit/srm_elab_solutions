    {

      scanf("%d",&n);
      for(c=0;c<n;c++)
      scanf("%d",&array[c]);
      scanf("%d",&value);
      scanf("%d",&position);
      for(c=n-1;c>=position;c--)
       array[c+1]=array[c];
       array[position]=value;

    }

**Q. 18: Occurrences of Characters**

    #include<stdio.h>
    #include<string.h>
    int main()
    {
      char s[200];
      int t,len,arr[26],i,f;
      scanf("%d",&t);
      while(t--)
      {
        for(i=0;i<26;i++) arr[i] = 0;
        scanf("%s",s);
        len = strlen(s);
        for(i=0;i<len;i++)
        {
          arr[s[i]- 'a']++;
        }
        f=0;
        for(i=0;i<26;i++)
        {
          if(arr[i] == len - arr[i]) { f = 1; break;}
        }
        if(f == 1) printf("YES\n");
        else printf("NO\n");
      }
      return 0;
    }
